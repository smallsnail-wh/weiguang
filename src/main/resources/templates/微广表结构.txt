CREATE TABLE `weixin_user` (
  `openid` varchar(225) NOT NULL,
  `nickname` varchar(225) NOT NULL,
  `sex` int(1) DEFAULT NULL,
  `province` varchar(225) DEFAULT NULL,
  `city` varchar(225) DEFAULT NULL,
  `country` varchar(225) DEFAULT NULL,
  `headimgurl` text,
  `unionid` varchar(225) DEFAULT NULL,
  `userid` int(10) DEFAULT NULL,
  PRIMARY KEY (`openid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='微信用户信息表';

CREATE TABLE `sys_user` (
  `id` int(10) NOT NULL AUTO_INCREMENT COMMENT 'id',
  `name` varchar(50) DEFAULT NULL COMMENT '姓名',
  `password` varchar(225) DEFAULT NULL COMMENT '密码',
  `email` varchar(225) DEFAULT NULL COMMENT '邮箱',
  `headimgurl` text COMMENT '头像',
  `level` int(1) DEFAULT NULL COMMENT '等级',
  `weixin_id` varchar(225) DEFAULT NULL COMMENT '微信的openid',
  `mobile` varchar(225) DEFAULT NULL COMMENT '手机号',
  `money` double DEFAULT '0' COMMENT '账号余额',
  `invite_code` varchar(225) NOT NULL COMMENT '邀请码',
  `create_time` varchar(225) NOT NULL COMMENT '用户注册时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=22 DEFAULT CHARSET=utf8 COMMENT='用户';
